%%%
% 【スタイル調整】
%%%
% ・タイトルの上につく余白をなくす
% ・見出しの上下につく余白を0.5行分のサイズにする
% ・表のキャプションの下につく余白を0.5行分のサイズにする
% ・脚注番号の前にスターを付ける（e.g. *1）
% ・箇条書きの行間を詰める
% ・図のセンタリング
% ・表のセンタリング
% ・ハイパーリンク

\makeatletter
  \renewcommand{\@maketitle}{
    \begin{center}
      {\huge \@title \par}
      \vskip 1.5em
      {\large \lineskip .5em
        \begin{tabular}[t]{c}
          \@author
        \end{tabular}
        \par
      }
      \vskip 1em
      {\large \@date}
    \end{center}
    \par
    \vskip 1.5em
  }
\makeatother
%
\makeatletter

}
\renewcommand{\subsubsection}{
  \@startsection{subsubsection}{3}{\z@}{0.5\Cvs}{0.5\Cvs}{\normalsize\bfseries}
}
